// ================== 网络模式切换 ==================
void setupWiFiAP() {
  WiFi.disconnect(true);
  WiFi.mode(WIFI_AP);
  WiFi.softAP(AP_SSID, AP_PASS);
  isAPMode = true;
  Serial.println("\n[WiFi] 已切换至 AP 模式 (热点)");
  Serial.print("[WiFi] 热点 IP: ");
  Serial.println(WiFi.softAPIP());
}

void setupWiFiSTA(String ssid, String pass) {
  WiFi.disconnect(true);
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid.c_str(), pass.c_str());
  isAPMode = false;
  
  Serial.printf("\n[WiFi] 正在连接至路由器: %s\n", ssid.c_str());
  
  int timeout = 0;
  while (WiFi.status() != WL_CONNECTED && timeout < 20) {
    delay(500);
    Serial.print(".");
    timeout++;
  }
  
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n[WiFi] STA 连接成功！");
    Serial.print("[WiFi] 局域网 IP: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("\n[WiFi] 连接失败！自动退回 AP 模式...");
    setupWiFiAP(); 
  }
}

// ================== 【核心解药】：独立文本/WiFi指令执行器 ==================
void processPcCommand(String cmd) {
  cmd.trim();
  if (cmd.length() == 0) return;

  // 1. 白名单 A：如果是完整的 JSON，透传给 Mesh 板
  if (cmd.startsWith("{") && cmd.endsWith("}")) {
    Serial.println("\n[PC -> UART1 (JSON透传)] " + cmd);
    Serial1.println(cmd);
    return;
  }

  // 2. 转换为小写，匹配本地指令
  String cmdLower = cmd;
  cmdLower.toLowerCase();

  // 3. 白名单 B：处理本地 WiFi 控制指令
  if (cmdLower == "wifi ap") {
    setupWiFiAP();
  } 
  else if (cmdLower == "wifi ip") {
    Serial.println("\n----- 当前网络状态 -----");
    if (isAPMode) {
      Serial.print("模式: AP (ESP32发出的热点)\nIP地址: "); Serial.println(WiFi.softAPIP());
    } else {
      Serial.print("模式: STA (连接家里路由器)\nIP地址: "); Serial.println(WiFi.localIP());
    }
    Serial.println("------------------------");
  }
  else if (cmdLower.startsWith("wifi sta ")) {
    int spaceIdx = cmd.indexOf(' ', 9); 
    if (spaceIdx != -1) {
      String ssid = cmd.substring(9, spaceIdx);
      String pass = cmd.substring(spaceIdx + 1);
      ssid.trim(); pass.trim();
      setupWiFiSTA(ssid, pass);
    } else {
      Serial.println("\n[错误] 请用空格隔开账号密码！例如: wifi sta MyHome 12345678");
    }
  } 
  else {
    // 4. 拦截区：既不是 JSON，也不是合法的 wifi 指令
    // 彻底砍掉之前的 Serial1.println(cmd); 绝不透传！
    Serial.println("\n[拦截] 无效指令或非 JSON 格式，已丢弃: " + cmd);
  }
}

// ================== WebSocket 数据转发 ==================
void onEvent(AsyncWebSocket* wsServer, AsyncWebSocketClient* client, AwsEventType type, void* arg, uint8_t* data, size_t len) {
  (void)wsServer;
  if (type == WS_EVT_CONNECT) {
    Serial.printf("[WS] 网页端接入. ID: %u\n", client->id());
  } else if (type == WS_EVT_DISCONNECT) {
    Serial.printf("[WS] 网页端断开.\n");
  } else if (type == WS_EVT_DATA) {
    AwsFrameInfo* info = (AwsFrameInfo*)arg;
    if (info->final && info->index == 0 && info->len == len && info->opcode == WS_TEXT) {
      String msg((const char*)data, len);
      Serial.printf("[Web -> Mesh] %s\n", msg.c_str());
      Serial1.println(msg); 
    }
  }
}

void setup() {
  Serial.setRxBufferSize(1024);
  Serial1.setRxBufferSize(2048);
  Serial.begin(115200);
  
  pcBuf.reserve(512);
  serverBuf.reserve(2048);

  pinMode(UART1_RX_PIN, INPUT_PULLUP);
  Serial1.begin(115200, SERIAL_8N1, UART1_RX_PIN, UART1_TX_PIN);
  delay(50);
  Serial1.flush();
  Serial1.print("\n\n\n");

  Serial.println("\n\n====================================");
  Serial.println("  Embedded Web Dashboard Gateway (Ultra-Fast) ");
  Serial.println("====================================");

  setupWiFiAP();

  ws.onEvent(onEvent);
  server.addHandler(&ws);

  server.on("/", HTTP_GET, [](AsyncWebServerRequest* request) {
    request->send_P(200, "text/html", EMBEDDED_CONSOLE_HTML);
  });
  server.onNotFound([](AsyncWebServerRequest* request) {
    request->redirect("/");
  });

  server.begin();
  Serial.println("[HTTP] Web Server 准备就绪!");
}

void loop() {
  uint32_t now = millis();
  ws.cleanupClients();

  // ================= 极速串口解析 =================
  while (Serial.available()) {
    char c = Serial.read();
    lastPcInputTime = now; // 刷新最后输入时间
    
    // 只接收可见字符和空格
    if (c >= 32 && c <= 126) {
      pcBuf += c;
    }

    // 动作A：检测到 JSON 闭括号，瞬间秒发
    if (c == '}') {
      int firstBrace = pcBuf.indexOf('{');
      if (firstBrace != -1) {
        String jsonStr = pcBuf.substring(firstBrace);
        Serial.println("\n[PC -> UART1 (JSON)] " + jsonStr);
        Serial1.println(jsonStr);
        pcBuf = ""; 
        continue;   
      }
    }

    // 动作B：物理回车触发
    if (c == '\n' || c == '\r') {
      if (pcBuf.length() > 0) {
        processPcCommand(pcBuf);
        pcBuf = ""; 
      }
    }
    
    if (pcBuf.length() > 512) pcBuf = ""; // 防爆栈
  }

  // ================= 自动回车 (Idle Timeout) =================
  // 如果缓冲区有数据，且 0.5秒内 没有新字符进来，直接判定输入完毕并执行！
  if (pcBuf.length() > 0 && (now - lastPcInputTime > 500)) {
    processPcCommand(pcBuf);
    pcBuf = "";
  }
  // ===========================================================

  // ================= 接收 Mesh 数据 =================
  while (Serial1.available()) {
    char c = Serial1.read();
    if (c == '\n' || c == '\r') {
      if (serverBuf.length() > 0) {
        lastServerRxTime = now;
        
        StaticJsonDocument<512> doc;
        DeserializationError err = deserializeJson(doc, serverBuf);
        
        if (!err && doc["ack"] == "ping") {
          if (!serverConnected) {
            serverConnected = true;
            Serial.println("\n[System] UART1 成功与 Mesh 网关握手!");
            Serial1.println("{\"cmd\":\"list\"}");
          }
        } else {
          ws.textAll(serverBuf);
        }
        serverBuf = "";
      }
    } else {
      if ((c >= 32 && c <= 126) || c == '{' || c == '}') serverBuf += c;
    }
    if (serverBuf.length() > 2048) serverBuf = "";
  }

  // 心跳维持
  uint32_t pingInterval = serverConnected ? 10000 : 2000;
  if (now - lastPingTime > pingInterval) {
    Serial1.println("{\"cmd\":\"ping\"}");
    lastPingTime = now;
  }
  if (serverConnected && (now - lastServerRxTime > 15000)) {
    serverConnected = false;
    Serial.println("\n[System] 警告: 与 Mesh 网关失去连接...");
  }

// ================== 【核心解药】：独立文本/WiFi指令执行器 ==================
void processPcCommand(String cmd) {
  cmd.trim();
  if (cmd.length() == 0) return;

  // 1. 白名单 A：如果是完整的 JSON，透传给 Mesh 板
  if (cmd.startsWith("{") && cmd.endsWith("}")) {
    Serial.println("\n[PC -> UART1 (JSON透传)] " + cmd);
    Serial1.println(cmd);
    return;
  }

  // 2. 转换为小写，匹配本地指令
  String cmdLower = cmd;
  cmdLower.toLowerCase();

  // 3. 白名单 B：处理本地 WiFi 控制指令
  if (cmdLower == "wifi ap") {
    setupWiFiAP();
  } 
  else if (cmdLower == "wifi ip") {
    Serial.println("\n----- 当前网络状态 -----");
    if (isAPMode) {
      Serial.print("模式: AP (ESP32发出的热点)\nIP地址: "); Serial.println(WiFi.softAPIP());
    } else {
      Serial.print("模式: STA (连接家里路由器)\nIP地址: "); Serial.println(WiFi.localIP());
    }
    Serial.println("------------------------");
  }
  else if (cmdLower.startsWith("wifi sta ")) {
    int spaceIdx = cmd.indexOf(' ', 9); 
    if (spaceIdx != -1) {
      String ssid = cmd.substring(9, spaceIdx);
      String pass = cmd.substring(spaceIdx + 1);
      ssid.trim(); pass.trim();
      setupWiFiSTA(ssid, pass);
    } else {
      Serial.println("\n[错误] 请用空格隔开账号密码！例如: wifi sta MyHome 12345678");
    }
  } 
  else {
    // 4. 拦截区：既不是 JSON，也不是合法的 wifi 指令
    // 彻底砍掉之前的 Serial1.println(cmd); 绝不透传！
    Serial.println("\n[拦截] 无效指令或非 JSON 格式，已丢弃: " + cmd);
  }
}

// ================== WebSocket 数据转发 ==================
void onEvent(AsyncWebSocket* wsServer, AsyncWebSocketClient* client, AwsEventType type, void* arg, uint8_t* data, size_t len) {
  (void)wsServer;
  if (type == WS_EVT_CONNECT) {
    Serial.printf("[WS] 网页端接入. ID: %u\n", client->id());
  } else if (type == WS_EVT_DISCONNECT) {
    Serial.printf("[WS] 网页端断开.\n");
  } else if (type == WS_EVT_DATA) {
    AwsFrameInfo* info = (AwsFrameInfo*)arg;
    if (info->final && info->index == 0 && info->len == len && info->opcode == WS_TEXT) {
      String msg((const char*)data, len);
      Serial.printf("[Web -> Mesh] %s\n", msg.c_str());
      Serial1.println(msg); 
    }
  }
}

void setup() {
  Serial.setRxBufferSize(1024);
  Serial1.setRxBufferSize(2048);
  Serial.begin(115200);
  
  pcBuf.reserve(512);
  serverBuf.reserve(2048);

  pinMode(UART1_RX_PIN, INPUT_PULLUP);
  Serial1.begin(115200, SERIAL_8N1, UART1_RX_PIN, UART1_TX_PIN);
  delay(50);
  Serial1.flush();
  Serial1.print("\n\n\n");

  Serial.println("\n\n====================================");
  Serial.println("  Embedded Web Dashboard Gateway (Ultra-Fast) ");
  Serial.println("====================================");

  setupWiFiAP();

  ws.onEvent(onEvent);
  server.addHandler(&ws);

  server.on("/", HTTP_GET, [](AsyncWebServerRequest* request) {
    request->send_P(200, "text/html", EMBEDDED_CONSOLE_HTML);
  });
  server.onNotFound([](AsyncWebServerRequest* request) {
    request->redirect("/");
  });

  server.begin();
  Serial.println("[HTTP] Web Server 准备就绪!");
}

void loop() {
  uint32_t now = millis();
  ws.cleanupClients();

  // ================= 极速串口解析 =================
  while (Serial.available()) {
    char c = Serial.read();
    lastPcInputTime = now; // 刷新最后输入时间
    
    // 只接收可见字符和空格
    if (c >= 32 && c <= 126) {
      pcBuf += c;
    }

    // 动作A：检测到 JSON 闭括号，瞬间秒发
    if (c == '}') {
      int firstBrace = pcBuf.indexOf('{');
      if (firstBrace != -1) {
        String jsonStr = pcBuf.substring(firstBrace);
        Serial.println("\n[PC -> UART1 (JSON)] " + jsonStr);
        Serial1.println(jsonStr);
        pcBuf = ""; 
        continue;   
      }
    }

    // 动作B：物理回车触发
    if (c == '\n' || c == '\r') {
      if (pcBuf.length() > 0) {
        processPcCommand(pcBuf);
        pcBuf = ""; 
      }
    }
    
    if (pcBuf.length() > 512) pcBuf = ""; // 防爆栈
  }

  // ================= 自动回车 (Idle Timeout) =================
  // 如果缓冲区有数据，且 0.5秒内 没有新字符进来，直接判定输入完毕并执行！
  if (pcBuf.length() > 0 && (now - lastPcInputTime > 500)) {
    processPcCommand(pcBuf);
    pcBuf = "";
  }
  // ===========================================================

  // ================= 接收 Mesh 数据 =================
  while (Serial1.available()) {
    char c = Serial1.read();
    if (c == '\n' || c == '\r') {
      if (serverBuf.length() > 0) {
        lastServerRxTime = now;
        
        StaticJsonDocument<512> doc;
        DeserializationError err = deserializeJson(doc, serverBuf);
        
        if (!err && doc["ack"] == "ping") {
          if (!serverConnected) {
            serverConnected = true;
            Serial.println("\n[System] UART1 成功与 Mesh 网关握手!");
            Serial1.println("{\"cmd\":\"list\"}");
          }
        } else {
          ws.textAll(serverBuf);
        }
        serverBuf = "";
      }
    } else {
      if ((c >= 32 && c <= 126) || c == '{' || c == '}') serverBuf += c;
    }
    if (serverBuf.length() > 2048) serverBuf = "";
  }

  // 心跳维持
  uint32_t pingInterval = serverConnected ? 10000 : 2000;
  if (now - lastPingTime > pingInterval) {
    Serial1.println("{\"cmd\":\"ping\"}");
    lastPingTime = now;
  }
  if (serverConnected && (now - lastServerRxTime > 15000)) {
    serverConnected = false;
    Serial.println("\n[System] 警告: 与 Mesh 网关失去连接...");
  }

    Serial.println("\n[拦截] 无效指令或非 JSON 格式，已丢弃: " + cmd);
  }
}

// ================== WebSocket 数据转发 ==================
void onEvent(AsyncWebSocket* wsServer, AsyncWebSocketClient* client, AwsEventType type, void* arg, uint8_t* data, size_t len) {
  (void)wsServer;
  if (type == WS_EVT_CONNECT) {
    Serial.printf("[WS] 网页端接入. ID: %u\n", client->id());
  } else if (type == WS_EVT_DISCONNECT) {
    Serial.printf("[WS] 网页端断开.\n");
  } else if (type == WS_EVT_DATA) {
    AwsFrameInfo* info = (AwsFrameInfo*)arg;
    if (info->final && info->index == 0 && info->len == len && info->opcode == WS_TEXT) {
      String msg((const char*)data, len);
      Serial.printf("[Web -> Mesh] %s\n", msg.c_str());
      Serial1.println(msg); 
    }
  }
}

void setup() {
  Serial.setRxBufferSize(1024);
  Serial1.setRxBufferSize(2048);
  Serial.begin(115200);
  
  pcBuf.reserve(512);
  serverBuf.reserve(2048);

  pinMode(UART1_RX_PIN, INPUT_PULLUP);
  Serial1.begin(115200, SERIAL_8N1, UART1_RX_PIN, UART1_TX_PIN);
  delay(50);
  Serial1.flush();
  Serial1.print("\n\n\n");

  Serial.println("\n\n====================================");
  Serial.println("  Embedded Web Dashboard Gateway (Ultra-Fast) ");
  Serial.println("====================================");

  setupWiFiAP();

  ws.onEvent(onEvent);
  server.addHandler(&ws);

  server.on("/", HTTP_GET, [](AsyncWebServerRequest* request) {
    request->send_P(200, "text/html", EMBEDDED_CONSOLE_HTML);
  });
  server.onNotFound([](AsyncWebServerRequest* request) {
    request->redirect("/");
  });

  server.begin();
  Serial.println("[HTTP] Web Server 准备就绪!");
}

void loop() {
  uint32_t now = millis();
  ws.cleanupClients();

  // ================= 极速串口解析 =================
  while (Serial.available()) {
    char c = Serial.read();
    lastPcInputTime = now; // 刷新最后输入时间
    
    // 只接收可见字符和空格
    if (c >= 32 && c <= 126) {
      pcBuf += c;
    }

    // 动作A：检测到 JSON 闭括号，瞬间秒发
    if (c == '}') {
      int firstBrace = pcBuf.indexOf('{');
      if (firstBrace != -1) {
        String jsonStr = pcBuf.substring(firstBrace);
        Serial.println("\n[PC -> UART1 (JSON)] " + jsonStr);
        Serial1.println(jsonStr);
        pcBuf = ""; 
        continue;   
      }
    }

    // 动作B：物理回车触发
    if (c == '\n' || c == '\r') {
      if (pcBuf.length() > 0) {
        processPcCommand(pcBuf);
        pcBuf = ""; 
      }
    }
    
    if (pcBuf.length() > 512) pcBuf = ""; // 防爆栈
  }

  // ================= 自动回车 (Idle Timeout) =================
  // 如果缓冲区有数据，且 0.5秒内 没有新字符进来，直接判定输入完毕并执行！
  if (pcBuf.length() > 0 && (now - lastPcInputTime > 500)) {
    processPcCommand(pcBuf);
    pcBuf = "";
  }
  // ===========================================================

  // ================= 接收 Mesh 数据 =================
  while (Serial1.available()) {
    char c = Serial1.read();
    if (c == '\n' || c == '\r') {
      if (serverBuf.length() > 0) {
        lastServerRxTime = now;
        
        StaticJsonDocument<512> doc;
        DeserializationError err = deserializeJson(doc, serverBuf);
        
        if (!err && doc["ack"] == "ping") {
          if (!serverConnected) {
            serverConnected = true;
            Serial.println("\n[System] UART1 成功与 Mesh 网关握手!");
            Serial1.println("{\"cmd\":\"list\"}");
          }
        } else {
          ws.textAll(serverBuf);
        }
        serverBuf = "";
      }
    } else {
      if ((c >= 32 && c <= 126) || c == '{' || c == '}') serverBuf += c;
    }
    if (serverBuf.length() > 2048) serverBuf = "";
  }

  // 心跳维持
  uint32_t pingInterval = serverConnected ? 10000 : 2000;
  if (now - lastPingTime > pingInterval) {
    Serial1.println("{\"cmd\":\"ping\"}");
    lastPingTime = now;
  }
  if (serverConnected && (now - lastServerRxTime > 15000)) {
    serverConnected = false;
    Serial.println("\n[System] 警告: 与 Mesh 网关失去连接...");
  }
}
