// ================== Network Mode Switching (with NVS Memory) ==================
void setupWiFiAP() {
  WiFi.disconnect(true);
  WiFi.mode(WIFI_AP);
  WiFi.softAP(AP_SSID, AP_PASS);
  isAPMode = true;
  Serial.println("\n[WiFi] Switched to AP Mode (Hotspot)");
  Serial.print("[WiFi] Hotspot IP: ");
  Serial.println(WiFi.softAPIP());
}

void setupWiFiSTA(String ssid, String pass, bool save = true) {
  WiFi.disconnect(true);
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid.c_str(), pass.c_str());
  isAPMode = false;
  
  Serial.printf("\n[WiFi] Attempting to connect to: %s\n", ssid.c_str());
  
  int timeout = 0;
  while (WiFi.status() != WL_CONNECTED && timeout < 20) {
    delay(500);
    Serial.print(".");
    timeout++;
  }
  
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n[WiFi] STA Connection Successful!");
    Serial.print("[WiFi] LAN IP: ");
    Serial.println(WiFi.localIP());

    // Save to Flash only when manually commanded and connected successfully
    if (save) {
      preferences.begin("wifi-config", false);
      preferences.putString("ssid", ssid);
      preferences.putString("pass", pass);
      preferences.end();
      Serial.println("[NVS] WiFi configuration saved permanently.");
    }
  } else {
    Serial.println("\n[WiFi] Connection Failed! Falling back to AP Mode...");
    setupWiFiAP(); 
  }
}

// ================== Independent Text/WiFi Command Processor ==================
void processPcCommand(String cmd) {
  cmd.trim();
  if (cmd.length() == 0) return;

  // Transparently pass JSON to Mesh board
  if (cmd.startsWith("{") && cmd.endsWith("}")) {
    Serial.println("\n[PC -> UART1 (JSON Passthrough)] " + cmd);
    Serial1.println(cmd);
    return;
  }

  String cmdLower = cmd;
  cmdLower.toLowerCase();

  if (cmdLower == "wifi ap") {
    setupWiFiAP();
  } 
  else if (cmdLower == "wifi ip") {
    Serial.println("\n----- Current Network Status -----");
    if (isAPMode) {
      Serial.print("Mode: AP (Hotspot)\nIP: "); Serial.println(WiFi.softAPIP());
    } else {
      Serial.print("Mode: STA (External Network)\nIP: "); Serial.println(WiFi.localIP());
    }
    Serial.println("-----------------------------------");
  }
  else if (cmdLower.startsWith("wifi sta ")) {
    int spaceIdx = cmd.indexOf(' ', 9); 
    if (spaceIdx != -1) {
      String ssid = cmd.substring(9, spaceIdx);
      String pass = cmd.substring(spaceIdx + 1);
      ssid.trim(); pass.trim();
      setupWiFiSTA(ssid, pass, true); // Manual connection, set to save
    } else {
      Serial.println("\n[Error] Missing parameters! Usage: wifi sta SSID PASSWORD");
    }
  } 
  else if (cmdLower == "wifi forget") {
    preferences.begin("wifi-config", false);
    preferences.clear();
    preferences.end();
    Serial.println("\n[NVS] Saved WiFi information cleared.");
  }
  else {
    Serial.println("\n[Blocked] Invalid command or non-JSON format: " + cmd);
  }
}

// ================== WebSocket Data Forwarding ==================
void onEvent(AsyncWebSocket* wsServer, AsyncWebSocketClient* client, AwsEventType type, void* arg, uint8_t* data, size_t len) {
  (void)wsServer;
  if (type == WS_EVT_CONNECT) {
    Serial.printf("[WS] Client connected. ID: %u\n", client->id());
  } else if (type == WS_EVT_DISCONNECT) {
    Serial.printf("[WS] Client disconnected.\n");
  } else if (type == WS_EVT_DATA) {
    AwsFrameInfo* info = (AwsFrameInfo*)arg;
    if (info->final && info->index == 0 && info->len == len && info->opcode == WS_TEXT) {
      String msg((const char*)data, len);
      Serial.printf("[Web -> Mesh] %s\n", msg.c_str());
      Serial1.println(msg); 
    }
  }
}

void setup() {
  Serial.setRxBufferSize(1024);
  Serial1.setRxBufferSize(2048);
  Serial.begin(115200);
  
  pcBuf.reserve(512);
  serverBuf.reserve(2048);

  pinMode(UART1_RX_PIN, INPUT_PULLUP);
  Serial1.begin(115200, SERIAL_8N1, UART1_RX_PIN, UART1_TX_PIN);
  delay(50);
  Serial1.flush();
  Serial1.print("\n\n\n");

  Serial.println("\n\n====================================");
  Serial.println("  Embedded Web Gateway (NVS Ready) ");
  Serial.println("====================================");

  // --- NVS Read Logic ---
  preferences.begin("wifi-config", true); // Open in read-only mode
  String savedSSID = preferences.getString("ssid", "");
  String savedPASS = preferences.getString("pass", "");
  preferences.end();

  if (savedSSID != "" && savedSSID.length() > 0) {
    Serial.print("[NVS] Found saved WiFi: "); Serial.println(savedSSID);
    setupWiFiSTA(savedSSID, savedPASS, false); // Auto-connect, don't write to Flash again
  } else {
    setupWiFiAP(); 
  }

  ws.onEvent(onEvent);
  server.addHandler(&ws);

  server.on("/", HTTP_GET, [](AsyncWebServerRequest* request) {
    request->send_P(200, "text/html", EMBEDDED_CONSOLE_HTML);
  });
  server.onNotFound([](AsyncWebServerRequest* request) {
    request->redirect("/");
  });

  server.begin();
  Serial.println("[HTTP] Web Server Ready!");
}

void loop() {
  uint32_t now = millis();
  ws.cleanupClients();

  // ================= Fast Serial Parsing =================
  while (Serial.available()) {
    char c = Serial.read();
    lastPcInputTime = now; 
    
    if (c >= 32 && c <= 126) pcBuf += c;

    if (c == '}') {
      int firstBrace = pcBuf.indexOf('{');
      if (firstBrace != -1) {
        String jsonStr = pcBuf.substring(firstBrace);
        Serial.println("\n[PC -> UART1 (JSON)] " + jsonStr);
        Serial1.println(jsonStr);
        pcBuf = ""; 
        continue;   
      }
    }

    if (c == '\n' || c == '\r') {
      if (pcBuf.length() > 0) {
        processPcCommand(pcBuf);
        pcBuf = ""; 
      }
    }
    if (pcBuf.length() > 512) pcBuf = ""; 
  }

  // Auto-enter (Idle Timeout)
  if (pcBuf.length() > 0 && (now - lastPcInputTime > 500)) {
    processPcCommand(pcBuf);
    pcBuf = "";
  }

  // ================= Receive Mesh Data =================
  while (Serial1.available()) {
    char c = Serial1.read();
    if (c == '\n' || c == '\r') {
      if (serverBuf.length() > 0) {
        lastServerRxTime = now;
        StaticJsonDocument<512> doc;
        DeserializationError err = deserializeJson(doc, serverBuf);
        
        if (!err && doc["ack"] == "ping") {
          if (!serverConnected) {
            serverConnected = true;
            Serial.println("\n[System] UART1 Handshake with Mesh Gateway Successful!");
            Serial1.println("{\"cmd\":\"list\"}");
          }
        } else {
          ws.textAll(serverBuf);
        }
        serverBuf = "";
      }
    } else {
      if ((c >= 32 && c <= 126) || c == '{' || c == '}') serverBuf += c;
    }
    if (serverBuf.length() > 2048) serverBuf = "";
  }

  // Keep-alive Handshake
  uint32_t pingInterval = serverConnected ? 10000 : 2000;
  if (now - lastPingTime > pingInterval) {
    Serial1.println("{\"cmd\":\"ping\"}");
    lastPingTime = now;
  }
  if (serverConnected && (now - lastServerRxTime > 15000)) {
    serverConnected = false;
    Serial.println("\n[System] Warning: Connection to Mesh Gateway Lost...");
  }
}
